fonction nombre fibo(nombre count){
    nombre num;
    si(count == 0 || count == 1){
        num = count;
    }
    autre{
        num = appel fibo(count - 1) + appel fibo(count - 2);
    };
    rappel num;
}

essentiel{

    nombre x;

    analyse("Enter how many fibo values: ", x);

    nombre first = 0;
    nombre second = 0;
    pour(nombre i=0; i<x; i++){
        si(first == 0 && second == 0){
            second = 1;
            imprimer("1");
        }
        autre{
            nombre sum = first + second;
            first = second;
            second = sum;
            imprimer(", " + sum);
        };
    };
    imprimer("nextline");

    analyse("Recursive fibo: ", x);
    nombre rec;
    rec = appel fibo(x);
    imprimer("The " + x + "th Fibonacci number is " + rec);
}
